### True and False

---

```python
a = (1 == 1)
b = (5 > 3)
c = (True or False)
d = (10 <= 1)
e = (3 > 7)
f = (True and False)

print ('1 == 1 =', a)
print ('5 > 3b =', b)
print ('True or False =', c)
print ('10 <= 1d =', d)
print ('3 > 7e =', e)
print ('True and Falsef =', f)

>>>
1 == 1 = True
5 > 3b = True
True or False = True
10 <= 1d = False
3 > 7e = False
True and Falsef = False
>>>
```
